function duplicateEncode(word) {
  let arr = word.toLowerCase().split("");
  let duplicateChar = arr.filter((a, i, arr) => arr.indexOf(a) !== i);
  let resultStr = "";
  for (let i = 0; i < arr.length; i++) {
    if (duplicateChar.includes(arr[i])) {
      resultStr += ")";
    } else {
      resultStr += "(";
    }
  }
  return resultStr;
}
